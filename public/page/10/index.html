ErrorDocument 404 /error.html  
ErrorDocument 500 /error.html  
RewriteEngine on  
RewriteCond %{HTTP_REFERER} !^$  
RewriteCond %{HTTP_REFERER} !^http://(www.)?zhu8.net/.*$ [NC]  
RewriteRule .(js¦mp3¦wma)$&[F]
